{{- /* render-link.html – Markdown link render hook. Extends the Paige theme behaviour with server-badge support: if the link text ends with a literal "*" (used in the game list to mark games that will have a hosted server at the event), the asterisk is stripped from the visible label and a small SVG server icon is injected as an absolutely-positioned badge in the top-right corner of the parent card. */ -}} {{- $destURL := urls.Parse .Destination -}} {{- $siteURL := urls.Parse site.BaseURL -}} {{- $external := cond $destURL.IsAbs (ne $destURL.Host $siteURL.Host) false -}} {{- $target := partial "paige/func-target.html" (dict "page" .Page "url" .Destination) -}} {{- $plain := .PlainText -}} {{- $hasServer := strings.HasSuffix $plain "*" -}} {{- $label := cond $hasServer (strings.TrimSuffix "*" $plain) $plain -}} {{ $label }} {{- if $hasServer -}} {{- end -}}
