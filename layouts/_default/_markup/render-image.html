{{- /* render-image.html – Markdown image render hook. Resolution order for the image resource: 1. Page-bundle resource (local file alongside the content file) 2. Global asset (assets/ directory) 3. Remote URL looked up from data/games.yaml by filename stem Raster images are resized to at most 460 px wide and converted to WebP for performance. The output always carries loading="lazy", decoding="async", and intrinsic width/height to prevent Cumulative Layout Shift (CLS). SVG resources are served as-is (no raster processing). If the image cannot be resolved, a plain fallback is emitted so existing markdown that references external URLs still renders correctly. */ -}} {{- $alt := .PlainText -}} {{- $title := .Title -}} {{- $src := .Destination -}} {{- $page := .Page -}} {{- $resource := "" -}} {{- /* 1. Page-bundle resource */ -}} {{- with $page.Resources.Get $src -}} {{- $resource = . -}} {{- end -}} {{- /* 2. Global asset */ -}} {{- if not $resource -}} {{- with resources.Get $src -}} {{- $resource = . -}} {{- end -}} {{- end -}} {{- /* 3. Remote URL via data/games.yaml, keyed by the filename stem */ -}} {{- if not $resource -}} {{- $slug := path.Base $src | strings.TrimSuffix (path.Ext $src) -}} {{- with index site.Data.games $slug -}} {{- $opts := dict "headers" (dict "User-Agent" "Mozilla/5.0 (compatible; Hugo; +https://www.ctrl-alt-gg.hu/)") -}} {{- with try (resources.GetRemote . $opts) -}} {{- with .Err -}} {{- warnf "render-image: cannot fetch remote image for %q: %s" $src . -}} {{- else with .Value -}} {{- $resource = . -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Render */ -}} {{- if $resource -}} {{- $isSVG := in "svg" (lower $resource.MediaType.SubType) -}} {{- if $isSVG -}} {{- else -}} {{- $w := 460 -}} {{- if and (gt $resource.Width 0) (lt $resource.Width 460) -}} {{- $w = $resource.Width -}} {{- end -}} {{- $img := $resource.Resize (printf "%dx webp" $w) -}} {{- $img = fingerprint $img -}} {{- end -}} {{- else -}} {{- end -}}
